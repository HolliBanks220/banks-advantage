import { createClient } from "jsr:@supabase/supabase-js@2"; const CLIENT_ID = Deno.env.get("GOOGLE_CLIENT_ID")!; const CLIENT_SECRET = Deno.env.get("GOOGLE_CLIENT_SECRET")!; const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ); const CORS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "*", "Content-Type": "application/json", }; async function getAccessToken(agentId: string) { const { data } = await supabase .from("calendar_tokens") .select("refresh_token") .eq("agent_id", agentId) .maybeSingle(); if (!data) return null; const res = await fetch("https://oauth2.googleapis.com/token", { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded" }, body: new URLSearchParams({ client_id: CLIENT_ID, client_secret: CLIENT_SECRET, refresh_token: data.refresh_token, grant_type: "refresh_token", }), }); const t = await res.json(); return t.access_token || null; } Deno.serve(async (req) => { if (req.method === "OPTIONS") return new Response("ok", { headers: CORS }); const body = await req.json(); const token = await getAccessToken(body.agent_id); if (!token) { return new Response(JSON.stringify({ error: "not connected" }), { headers: CORS }); } if (body.action === "delete" && body.event_id) { await fetch("https://www.googleapis.com/calendar/v3/calendars/primary/events/" + body.event_id, { method: "DELETE", headers: { Authorization: "Bearer " + token }, }); return new Response(JSON.stringify({ ok: true }), { headers: CORS }); } const start = new Date(body.start); const end = new Date(start.getTime() + (body.minutes || 60) * 60000); const event = { summary: body.summary, description: body.description, start: { dateTime: start.toISOString() }, end: { dateTime: end.toISOString() }, }; const isUpdate = !!body.event_id; const url = isUpdate ? "https://www.googleapis.com/calendar/v3/calendars/primary/events/" + body.event_id : "https://www.googleapis.com/calendar/v3/calendars/primary/events"; const res = await fetch(url, { method: isUpdate ? "PATCH" : "POST", headers: { Authorization: "Bearer " + token, "Content-Type": "application/json", }, body: JSON.stringify(event), }); const out = await res.json(); return new Response(JSON.stringify({ event_id: out.id || null }), { headers: CORS }); });
